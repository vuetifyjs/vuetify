## Install

``` bash
# npm
npm install vuetify --save-dev
```

``` bash
# yarn
yarn add vuetify --dev
```

## Use

```javascript
import Vue from 'vue'
import Vuetify from 'vuetify'

Vue.use(Vuetify)
```
